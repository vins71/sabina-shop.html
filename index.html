async function submitOrder(){ if(_checkoutForWA){ sendWhatsApp(); return; } const nome = document.getElementById('f-nome').value.trim(); const cognome = document.getElementById('f-cognome').value.trim(); const email = document.getElementById('f-email').value.trim(); const telefono = document.getElementById('f-tel').value.trim(); const indirizzo= document.getElementById('f-indirizzo').value.trim(); const note = document.getElementById('f-note').value.trim(); if(!nome || !email){ alert('Inserisci almeno nome ed email.'); return; } const prodotti = cart.map(i => { const noteStr = i.note ? ` [${i.note}]` : ''; return `${i.name} x${i.qty}${noteStr}`; }).join(', '); const totale = cart.reduce((s, i) => s + (i.price || 0) * i.qty, 0); const btn = document.querySelector('.submit-order'); if(btn){ btn.disabled = true; btn.textContent = 'Invio in corso...'; } closeCheckout(); try { const res = await fetch('/api/order', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ nome, cognome, email, telefono, indirizzo, prodotti, totale, pagamento: selectedPayment, note }) }); const data = await res.json(); if(!res.ok) console.error('Errore salvataggio ordine:', data.error); } catch(err) { console.error('Impossibile contattare il server:', err); } const msgs = { sumup: 'Grazie ' + nome + '! Ti contatteremo al più presto per comunicarti le spese di spedizione e inviarti il link di pagamento SumUp.', bonifico: 'Grazie ' + nome + '! Ti invieremo i dati per il bonifico e le spese di spedizione via email.', consegna: 'Grazie ' + nome + '! Sabina ti contatterà per concordare giorno e orario di ritiro a Imperia.' }; document.getElementById('success-msg').textContent = msgs[selectedPayment] || 'Grazie ' + nome + '! Ordine ricevuto.'; document.getElementById('success-overlay').classList.add('open'); if(btn){ btn.disabled = false; btn.textContent = 'Conferma ordine'; } _checkoutForWA = false; cart = []; updateCartUI(); }
